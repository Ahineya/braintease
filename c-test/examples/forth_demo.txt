# Define factorial (iterative)
: FACTORIAL 
  DUP 1 > IF
    DUP 1 - 
    BEGIN
      DUP 1 > WHILE
      SWAP OVER * SWAP 1 -
    REPEAT
    DROP
  THEN
;
